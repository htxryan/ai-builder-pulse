1. **src/config.ts** — Zod-validated config loader (AUTO-04):
   ```typescript
   import { z } from 'zod';

   // Load .env in development only (CI uses GitHub Secrets)
   if (process.env.NODE_ENV !== 'production') {
     const dotenv = await import('dotenv');
     dotenv.config();
   }

   const envSchema = z.object({
     ANTHROPIC_API_KEY: z.string().min(1, 'ANTHROPIC_API_KEY is required'),
     BUTTONDOWN_API_KEY: z.string().min(1, 'BUTTONDOWN_API_KEY is required'),
     REDDIT_CLIENT_ID: z.string().min(1, 'REDDIT_CLIENT_ID is required'),
     REDDIT_CLIENT_SECRET: z.string().min(1, 'REDDIT_CLIENT_SECRET is required'),
     NODE_ENV: z.enum(['development', 'production', 'test']).default('production'),
   });

   export type Config = z.infer<typeof envSchema>;

   export function loadConfig(): Config {
     const result = envSchema.safeParse(process.env);
     if (!result.success) {
       const missing = result.error.issues.map(i => i.path.join('.')).join(', ');
       throw new Error(`Missing or invalid environment variables: ${missing}`);
     }
     return result.data;
   }
   ```

2. **src/models.ts** — Zod schemas for pipeline data:
   ```typescript
   import { z } from 'zod';

   export const RawItemSchema = z.object({
     source: z.enum(['hackernews', 'github-trending', 'reddit', 'rss']),
     title: z.string().min(1),
     url: z.string().url(),
     timestamp: z.coerce.date(),
     score: z.number().optional(),
     metadata: z.record(z.unknown()).optional(),
   });

   export type RawItem = z.infer<typeof RawItemSchema>;

   export const ScoredItemSchema = RawItemSchema.extend({
     relevanceScore: z.number().min(0).max(1),
     category: z.enum([
       'tools-launches',
       'model-releases',
       'techniques-patterns',
       'infrastructure-deployment',
       'notable-discussions',
       'think-pieces-analysis',
       'news-in-brief',
     ]),
     description: z.string().min(1),
   });

   export type ScoredItem = z.infer<typeof ScoredItemSchema>;

   export const DailyIssueSchema = z.object({
     date: z.string().regex(/^\d{4}-\d{2}-\d{2}$/),
     items: z.array(ScoredItemSchema),
     metadata: z.object({
       totalCandidates: z.number(),
       totalAccepted: z.number(),
       tokenCount: z.number().optional(),
       estimatedCost: z.number().optional(),
     }),
   });

   export type DailyIssue = z.infer<typeof DailyIssueSchema>;
   ```

   import { describe, it, expect, beforeEach, afterEach } from 'vitest';
   import { loadConfig } from '../src/config.js';

   describe('loadConfig', () => {
     const originalEnv = process.env;

     beforeEach(() => {
       process.env = { ...originalEnv };
       process.env.NODE_ENV = 'test';
     });

     afterEach(() => {
       process.env = originalEnv;
     });

     it('returns Config when all required env vars are set', () => {
       process.env.ANTHROPIC_API_KEY = 'sk-ant-test';
       process.env.BUTTONDOWN_API_KEY = 'btn-test';
       process.env.REDDIT_CLIENT_ID = 'reddit-id';
       process.env.REDDIT_CLIENT_SECRET = 'reddit-secret';

       const config = loadConfig();
       expect(config.ANTHROPIC_API_KEY).toBe('sk-ant-test');
       expect(config.BUTTONDOWN_API_KEY).toBe('btn-test');
       expect(config.REDDIT_CLIENT_ID).toBe('reddit-id');
       expect(config.REDDIT_CLIENT_SECRET).toBe('reddit-secret');
     });

     it('throws when ANTHROPIC_API_KEY is missing', () => {
       process.env.BUTTONDOWN_API_KEY = 'btn-test';
       process.env.REDDIT_CLIENT_ID = 'reddit-id';
       process.env.REDDIT_CLIENT_SECRET = 'reddit-secret';
       delete process.env.ANTHROPIC_API_KEY;

       expect(() => loadConfig()).toThrow('Missing or invalid environment variables');
     });

     it('throws when any env var is empty string', () => {
       process.env.ANTHROPIC_API_KEY = '';
       process.env.BUTTONDOWN_API_KEY = 'btn-test';
       process.env.REDDIT_CLIENT_ID = 'reddit-id';
       process.env.REDDIT_CLIENT_SECRET = 'reddit-secret';

       expect(() => loadConfig()).toThrow('Missing or invalid environment variables');
     });

     it('throws when multiple env vars are missing', () => {

       delete process.env.BUTTONDOWN_API_KEY;
       delete process.env.REDDIT_CLIENT_ID;
       delete process.env.REDDIT_CLIENT_SECRET;

       expect(() => loadConfig()).toThrow('Missing or invalid environment variables');

     it('defaults NODE_ENV to production when not set', () => {

       process.env.REDDIT_CLIENT_SECRET = 'reddit-secret';
       delete process.env.NODE_ENV;

       const config = loadConfig();
       expect(config.NODE_ENV).toBe('production');
